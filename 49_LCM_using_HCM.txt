#Lowest Common Multipe(LCM) using Highest Common Factor(HCF)
# LCM of 12 and 15 is 60
   
def hcf(a,b):
    while b:
        a,b=b,a%b
    return a
def lcm(a,b):
    cal=(a*b)//hcf(a,b)
    return cal

n=int(input("how many times do you want to check : "))
for i in range(n):
    x = int(input('Enter 1st number: '))
    y = int(input('Enter 2nd number: '))
    print(f"Highest Common Factor of {x} and {y} is : {hcf(x,y)}")
    print(f"Lowest Common Multiple of {x} and {y} is: {lcm(x,y)}")
    i=i+1
